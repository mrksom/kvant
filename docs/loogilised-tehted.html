2 == 2 # võrdub
## [1] TRUE
1 != 2 # ei võrdu
## [1] TRUE
2 >= 2 # suurem kui või võrdne
## [1] TRUE
1 < 2  # väiksem kui
## [1] TRUE
(3<6) | (6<3) # loogiline või
## [1] TRUE
(3<6) & (6<3) # loogiline ja
## [1] FALSE
